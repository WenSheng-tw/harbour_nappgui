nap_command_app(${EXAMPLE_NAME} "hboffice" NRC_NONE)

    find_library(LIB_NAME_${lib} NAMES ${lib} PATHS "${HARBOUR_LIB_PATH}")
    target_link_libraries(${EXAMPLE_NAME} ${LIB_NAME_${lib}})
endforeach()
